```javascript
        const salaryIncrease = targetSalary - currentSalary;
        const paybackPeriod = investment / (salaryIncrease / 12);
        return { paybackMonths: paybackPeriod, annualROI: salaryIncrease };
    },
    showPricingTransparency: true
};
```
